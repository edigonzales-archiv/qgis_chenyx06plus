-- Prueft ob Dreieckspunkte auf TSP liegen.
-- Leider noch zwei Funktionen fuer beide
-- Bezugsrahmen. -> SRID kann man aus Geometrie
-- rausholen und dann if/else...
CREATE OR REPLACE FUNCTION ch.triangle_on_tsp_lv03(triangle geometry) RETURNS integer AS $$
DECLARE
 count_row RECORD;
BEGIN
 IF triangle IS NULL
 THEN
  RETURN 3;
 ELSE
  SELECT count(*)::integer as count INTO count_row
  FROM
  (
   SELECT 1 as ogc_fid, ST_Intersection(ST_SnapToGrid(b.the_geom, 0.001), ST_SnapToGrid(a.the_geom_lv03, 0.001)) as the_geom
   FROM ch.tsp as a,
   (
    SELECT ST_Collect(geom) as the_geom
    FROM
    (
     SELECT (ST_DumpPoints(triangle)).geom
    ) as a
   ) as b
    WHERE a.the_geom_lv03 && b.the_geom
  ) as c
  WHERE geometrytype(the_geom) = 'POINT';
  RETURN count_row.count;
  
 END IF;
END;
$$ LANGUAGE plpgsql;

CREATE OR REPLACE FUNCTION ch.triangle_on_tsp_lv95(triangle geometry) RETURNS integer AS $$
DECLARE
 count_row RECORD;
BEGIN
 IF triangle IS NULL
 THEN
  RETURN 3;
 ELSE
  SELECT count(*)::integer as count INTO count_row
  FROM
  (
   SELECT 1 as ogc_fid, ST_Intersection(ST_SnapToGrid(b.the_geom, 0.001), ST_SnapToGrid(a.the_geom_lv95, 0.001)) as the_geom
   FROM ch.tsp as a,
   (
    SELECT ST_Collect(geom) as the_geom
    FROM
    (
     SELECT (ST_DumpPoints(triangle)).geom
    ) as a
   ) as b
    WHERE a.the_geom_lv95 && b.the_geom
  ) as c
  WHERE geometrytype(the_geom) = 'POINT';
  RETURN count_row.count;
  
 END IF;
END;
$$ LANGUAGE plpgsql;

-- Daten in LV03- und LV95-Dreieckstabellen schreiben.
-- So *sollten* sie gleich definiert sein (also Punkt-
-- reihenfolge).
INSERT INTO ch.dreiecke_lv95 (nummer, typ, the_geom) 
SELECT dreieck_nummer, typ, ST_Reverse(ST_ForceRHR(ST_BuildArea(the_geom))) as the_geom 
FROM
(
 SELECT dreieck_nummer, typ, ST_AddPoint(the_geom, ST_PointN(the_geom,1)) as the_geom
 FROM
 (
  SELECT dreieck_nummer, typ, ST_LineFromMultiPoint(ST_Collect(the_geom)) as the_geom
  FROM
  (
   SELECT dreieck_nummer, typ, the_geom_lv95 as the_geom
   FROM ch.tsp as a,
   (
    SELECT *
    FROM ch.dreiecke_tsp
   ) as b
   WHERE b.tsp_nummer = a.nummer
  ) as c
  GROUP BY dreieck_nummer, typ
 ) as d
) as e
ORDER BY dreieck_nummer 

INSERT INTO ch.dreiecke_lv03 (nummer, typ, the_geom) 
SELECT dreieck_nummer, typ, ST_Reverse(ST_ForceRHR(ST_BuildArea(the_geom))) as the_geom

 SELECT dreieck_nummer, typ, ST_AddPoint(the_geom, ST_PointN(the_geom,1)) as the_geom

-- Daten in Lookup-Table schreiben.
INSERT INTO ch.dreiecke_tsp (tsp_nummer, dreieck_nummer) 
SELECT tsp_nummer, dreieck_nummer
FROM
(
 SELECT a.nummer as tsp_nummer, b.nummer as dreieck_nummer, ST_Intersection(ST_SnapToGrid(b.the_geom, 0.001), ST_SnapToGrid(a.the_geom_lv03, 0.001)) as the_geom
 FROM bern.tsp as a,
 (
   SELECT nummer, (ST_DumpPoints(the_geom)).geom as the_geom
   FROM
   (
     SELECT nummer, ST_RemovePoint(ST_Boundary(ST_Reverse(ST_ForceRHR(the_geom_lv03))),3) as the_geom FROM ch.dreiecke ORDER BY nummer 
   ) as c
 ) as b
 WHERE a.the_geom_lv03 && b.the_geom
